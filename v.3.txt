/**
 * ==============================================
 * KODE UTAMA - WEBHOOK HANDLER
 * ==============================================
 */

function doPost(e) {
  var spreadsheet = SpreadsheetApp.getActiveSpreadsheet();
  var logSheet = spreadsheet.getSheetByName("Log") || spreadsheet.insertSheet("Log");
  
  // Log request masuk secara detail
  logSheet.appendRow([new Date(), "Request masuk (raw): " + JSON.stringify(e)]);
  
  try {
    // 1. Ambil konfigurasi
    var configSheet = spreadsheet.getSheetByName("Konfigurasi") || createConfigSheet();
    var config = getConfigValues(configSheet);
    validateConfig(config);
    
    // 2. Ambil konfigurasi bot
    var botSheet = spreadsheet.getSheetByName("Bot") || createBotSheet();
    var botConfig = getBotConfigValues(botSheet);

    // 3. Parse data input
    var requestData = JSON.parse(e.postData.contents);
    var message = requestData.message || "";
    var senderNumber = requestData.from || "";
    logSheet.appendRow([new Date(), "Data parsed: " + JSON.stringify({message: message, senderNumber: senderNumber})]);

    // 4. Handle permintaan info akun
    if (message.toLowerCase().includes(botConfig.info_account_trigger.toLowerCase())) {
      return handleInfoAccountRequest(senderNumber, config, botConfig);
    }

    // 5. Handle konfirmasi user (Ya/Tidak)
    if (message.toLowerCase() === "ya" || message.toLowerCase() === "tidak") {
      return handleUserConfirmation(requestData, config, botConfig);
    }

    // 6. Cek apakah sedang menunggu pilihan afirmasi
    var tempAfirmasiData = getTempData(senderNumber + "_afirmasi");
    if (tempAfirmasiData && tempAfirmasiData.waitingAfirmasi) {
      // Jika pesan kosong, abaikan dan tidak lanjut ke validasi
      if (!message.trim()) {
        logSheet.appendRow([new Date(), "Pesan kosong diterima, tetap menunggu pilihan afirmasi"]);
        return createResponse("waiting_afirmasi", "Masih menunggu pilihan afirmasi");
      }
      return handleAfirmasiSelection(requestData, config, botConfig);
    }

    // 7. Parsing data pendaftaran (dengan error handling)
    var parsedData = parseRegistrationData(message);
    var hasRegistrationData = parsedData.name && parsedData.number; // Minimal harus ada nama dan nomor
    
    if (hasRegistrationData) {
      var name = parsedData.name || "";
      var email = parsedData.email || "";
      var number = parsedData.number || "";
      var nik = parsedData.nik ? "'" + parsedData.nik : ""; // Tambahkan apostrof untuk memastikan disimpan sebagai teks
      var nisn = parsedData.nisn ? "'" + parsedData.nisn : ""; // Tambahkan apostrof untuk memastikan disimpan sebagai teks
      var asalSekolah = parsedData.asalSekolah || "";
      var jenisKelamin = parsedData.jenisKelamin || "";
      var jenisAfirmasi = tempAfirmasiData ? tempAfirmasiData.jenisAfirmasi : "";
      
      // Validasi NIK (harus 16 digit angka)
      if (nik && !/^\d{16}$/.test(parsedData.nik)) {
        var errorMsg = botConfig.invalid_nik_message.replace("${admin_number}", config.admin_number);
        sendWhatsappMessage(senderNumber, errorMsg, config);
        logSheet.appendRow([new Date(), "NIK tidak valid: " + parsedData.nik]);
        return createResponse("invalid_nik", "NIK tidak valid");
      }
  
      // Validasi NISN (harus 10 digit angka)
      if (nisn && !/^\d{10}$/.test(parsedData.nisn)) {
        var errorMsg = botConfig.invalid_nisn_message.replace("${admin_number}", config.admin_number);
        sendWhatsappMessage(senderNumber, errorMsg, config);
        logSheet.appendRow([new Date(), "NISN tidak valid: " + parsedData.nisn]);
        return createResponse("invalid_nisn", "NISN tidak valid");
      }

      // Log hasil parsing
      logSheet.appendRow([new Date(), `Data parsed: Nama=${name}, Email=${email}, Nomor=${number}, Jenis Kelamin=${jenisKelamin}, NIK=${nik}, NISN=${nisn}, Asal Sekolah=${asalSekolah}, Jenis Afirmasi=${jenisAfirmasi}`]);

      // 8. Format nomor WA (dengan validasi ketat)
      var formattedNumber = formatNomorIndonesia(number);
      if (!formattedNumber) {
        throw new Error("Nomor WhatsApp tidak valid");
      }

      // 9. Validasi nomor WhatsApp
      if (!isWhatsappNumberValid(formattedNumber, config)) {
        var errorMsg = botConfig.invalid_number_message
          .replace("${formattedNumber}", formattedNumber)
          .replace("${admin_number}", config.admin_number);
        
        sendWhatsappMessage(senderNumber, errorMsg, config);
        logSheet.appendRow([new Date(), "Nomor tidak valid: " + formattedNumber]);
        return createResponse("invalid_number", "Nomor tidak valid");
      }

      // 10. Jika nomor valid, lanjutkan proses pendaftaran
      var pendaftaranSheet = spreadsheet.getSheetByName("Pendaftaran") || createPendaftaranSheet();
      var duplicateData = findDuplicateData(pendaftaranSheet, formattedNumber, nik);

      // Cek duplikasi NIK
      if (duplicateData && duplicateData.type === 'nik') {
        var errorMsg = botConfig.duplicate_nik_message
          .replace("${nik}", nik.replace(/^'/, ''))
          .replace("${name}", duplicateData.name)
          .replace("${number}", duplicateData.number)
          .replace("${admin_number}", config.admin_number);
        
        sendWhatsappMessage(senderNumber, errorMsg, config);
        logSheet.appendRow([new Date(), "NIK sudah terdaftar: " + nik]);
        return createResponse("duplicate_nik", "NIK sudah terdaftar");
      }

      // Cek duplikasi nomor WA
      if (duplicateData && duplicateData.type === 'number') {
        // Simpan data sementara untuk konfirmasi
        saveTempData(senderNumber, { 
          name: name, 
          email: email, 
          number: formattedNumber,
          jenisKelamin: jenisKelamin,
          nik: nik,
          nisn: nisn,
          asalSekolah: asalSekolah,
          jenisAfirmasi: jenisAfirmasi
        });
        
        // Kirim pesan konfirmasi update
        var confirmMsg = buildDuplicateMessage(duplicateData, {name, email, jenisKelamin, nik, nisn, asalSekolah, jenisAfirmasi}, botConfig);
        sendWhatsappMessage(senderNumber, confirmMsg, config);
        logSheet.appendRow([new Date(), "Menunggu konfirmasi update dari: " + senderNumber]);
        return createResponse("waiting_confirmation", "Menunggu konfirmasi user");
      } else {
        // Simpan data baru
        var generatedParticipant = saveToSheet(pendaftaranSheet, name, email, formattedNumber, jenisKelamin, nik, nisn, asalSekolah, jenisAfirmasi, config);
        
        // Kirim konfirmasi sukses
        var successMsg = generateReplyMessage(name, email, formattedNumber, jenisKelamin, generatedParticipant, nik, nisn, asalSekolah, jenisAfirmasi, config, botConfig);
        sendWhatsappMessage(formattedNumber, successMsg, config);
        
        // Hapus data afirmasi sementara
        clearTempData(senderNumber + "_afirmasi");
        
        logSheet.appendRow([new Date(), "Data baru disimpan: " + formattedNumber]);
        return createResponse("success", "Pendaftaran berhasil");
      }
    }
      // 11. Jika bukan data pendaftaran, cek trigger keyword
    else if (isTriggerMatch(message, botConfig.trigger_keyword)) {
      // Kirim pilihan jenis afirmasi
      var afirmasiMsg = botConfig.afirmasi_selection_message;
      sendWhatsappMessage(senderNumber, afirmasiMsg, config);
      
      // Simpan status menunggu pilihan afirmasi
      saveTempData(senderNumber + "_afirmasi", {
        waitingAfirmasi: true,
        timestamp: new Date().getTime() // Tambahkan timestamp untuk timeout
      });
      
      logSheet.appendRow([new Date(), "Trigger keyword terdeteksi, mengirim pilihan jenis afirmasi"]);
      return createResponse("afirmasi_selection", "Menunggu pilihan jenis afirmasi");
    } else {
      logSheet.appendRow([new Date(), "Pesan diabaikan: Bukan data pendaftaran atau trigger"]);
      return createResponse("ignored", "Pesan tidak dikenali");
    }

  } catch (error) {
    logSheet.appendRow([new Date(), "ERROR: " + error.message + "\nStack: " + error.stack]);
    return createResponse("error", "Terjadi kesalahan: " + error.message);
  }
}

/**
 * Handle pemilihan jenis afirmasi
 */
function handleAfirmasiSelection(data, config, botConfig) {
  var spreadsheet = SpreadsheetApp.getActiveSpreadsheet();
  var logSheet = spreadsheet.getSheetByName("Log");
  var senderNumber = data.from;
  var message = data.message.trim().toUpperCase();
  
  var validAfirmasi = ["KETM", "BINA LINGKUNGAN"];
  
  // Cek timeout (5 menit)
  var tempData = getTempData(senderNumber + "_afirmasi");
  var now = new Date().getTime();
  if (now - tempData.timestamp > 300000) { // 5 menit dalam milidetik
    clearTempData(senderNumber + "_afirmasi");
    logSheet.appendRow([new Date(), "Sesi pemilihan afirmasi timeout untuk: " + senderNumber]);
    sendWhatsappMessage(senderNumber, "Sesi pemilihan jenis afirmasi telah berakhir. Silakan mulai lagi dengan mengirim keyword pendaftaran.", config);
    return createResponse("afirmasi_timeout", "Sesi pemilihan afirmasi timeout");
  }
  
  if (validAfirmasi.includes(message)) {
    // Simpan pilihan afirmasi
    saveTempData(senderNumber + "_afirmasi", {
      jenisAfirmasi: message,
      waitingAfirmasi: false,
      timestamp: now
    });
    
    // Kirim petunjuk pendaftaran
    sendWhatsappMessage(senderNumber, botConfig.trigger_response, config);
    logSheet.appendRow([new Date(), "Jenis afirmasi dipilih: " + message + " oleh: " + senderNumber]);
    return createResponse("afirmasi_selected", "Jenis afirmasi dipilih: " + message);
  } else {
    // Kirim pesan error jika pilihan tidak valid
    var errorMsg = botConfig.invalid_afirmasi_message;
    sendWhatsappMessage(senderNumber, errorMsg, config);
    logSheet.appendRow([new Date(), "Pilihan afirmasi tidak valid: " + message]);
    return createResponse("invalid_afirmasi", "Pilihan afirmasi tidak valid");
  }
}

/**
 * Handle konfirmasi user (Ya/Tidak)
 */
function handleUserConfirmation(data, config, botConfig) {
  var spreadsheet = SpreadsheetApp.getActiveSpreadsheet();
  var logSheet = spreadsheet.getSheetByName("Log");
  var senderNumber = data.from;
  var message = data.message.toLowerCase();
  var tempData = getTempData(senderNumber);

  if (!tempData) {
    logSheet.appendRow([new Date(), "Tidak ada data temp untuk: " + senderNumber]);
    return createResponse("error", "Sesi konfirmasi tidak valid");
  }

  var pendaftaranSheet = spreadsheet.getSheetByName("Pendaftaran");
  var duplicateData = findDuplicateData(pendaftaranSheet, tempData.number, tempData.nik);

  if (message === "ya" && duplicateData) {
    // Update data lama dengan struktur baru
    pendaftaranSheet.getRange(duplicateData.row, 3, 1, 8).setValues([[ // 8 kolom karena TIMESTAMP dan NOMOR PENDAFTARAN tidak diupdate
      tempData.jenisAfirmasi || "", // JENIS AFIRMASI (kolom 3)
      tempData.name, // NAMA LENGKAP (kolom 4)
      tempData.number, // NOMOR WHATSAPP (kolom 5)
      tempData.jenisKelamin, // JENIS KELAMIN (kolom 6)
      tempData.email, // EMAIL (kolom 7)
      tempData.nik, // NIK (kolom 8) - disimpan sebagai teks
      tempData.nisn, // NISN (kolom 9) - disimpan sebagai teks
      tempData.asalSekolah // ASAL SEKOLAH (kolom 10)
    ]]);
    
    // Kirim notifikasi update
    var updateMsg = buildUpdateNotification(duplicateData, tempData, botConfig);
    sendWhatsappMessage(senderNumber, updateMsg, config);
    logSheet.appendRow([new Date(), "Data diperbarui: " + tempData.number]);
  } else {
    sendWhatsappMessage(senderNumber, botConfig.cancel_message, config);
    logSheet.appendRow([new Date(), "Pembaruan dibatalkan oleh user"]);
  }

  clearTempData(senderNumber);
  clearTempData(senderNumber + "_afirmasi");
  return createResponse("success", "Proses konfirmasi selesai");
}

/**
 * Handle permintaan info akun dari peserta
 */
function handleInfoAccountRequest(senderNumber, config, botConfig) {
  var spreadsheet = SpreadsheetApp.getActiveSpreadsheet();
  var logSheet = spreadsheet.getSheetByName("Log");
  var pendaftaranSheet = spreadsheet.getSheetByName("Pendaftaran");
  
  // Log permintaan info akun
  logSheet.appendRow([new Date(), "Permintaan info akun dari: " + senderNumber]);
  
  try {
    // Cari data berdasarkan nomor pengirim
    var data = findAccountData(pendaftaranSheet, senderNumber);
    
    if (data) {
      // Format tanggal pendaftaran
      var formattedDate = Utilities.formatDate(data.timestamp, Session.getScriptTimeZone(), "dd/MM/yyyy HH:mm");
      
      // Kirim info akun
      var infoMsg = buildInfoAccountMessage(data, formattedDate, config, botConfig);
      sendWhatsappMessage(senderNumber, infoMsg, config);
      logSheet.appendRow([new Date(), "Info akun dikirim ke: " + senderNumber]);
      return createResponse("info_sent", "Info akun dikirim");
    } else {
      // Kirim pesan data tidak ditemukan
      var notFoundMsg = botConfig.info_account_not_found.replace("${admin_number}", config.admin_number);
      sendWhatsappMessage(senderNumber, notFoundMsg, config);
      logSheet.appendRow([new Date(), "Data tidak ditemukan untuk: " + senderNumber]);
      return createResponse("not_found", "Data tidak ditemukan");
    }
  } catch (error) {
    logSheet.appendRow([new Date(), "ERROR info akun: " + error.message]);
    return createResponse("error", "Gagal memproses permintaan info akun");
  }
}

/**
 * Cari data akun berdasarkan nomor WA atau NIK
 */
function findAccountData(sheet, senderNumber) {
  if (!sheet || sheet.getLastRow() < 2) return null;
  
  var data = sheet.getRange(2, 1, sheet.getLastRow()-1, 10).getValues();
  for (var i = 0; i < data.length; i++) {
    var existingNumber = formatNomorIndonesia(data[i][4]); // Kolom NOMOR WHATSAPP sekarang di index 4
    
    // Cek berdasarkan nomor WA
    if (existingNumber === senderNumber) {
      return {
        timestamp: data[i][0],
        participant: data[i][1],
        jenisAfirmasi: data[i][2],
        name: data[i][3],
        number: existingNumber,
        jenisKelamin: data[i][5],
        email: data[i][6],
        nik: data[i][7] ? data[i][7].toString().replace(/^'/, '') : '',
        nisn: data[i][8] ? data[i][8].toString().replace(/^'/, '') : '',
        asalSekolah: data[i][9]
      };
    }
  }
  return null;
}

/**
 * Format nomor Indonesia dengan validasi ketat
 */
function formatNomorIndonesia(number) {
  if (!number) {
    Logger.log("Nomor tidak valid: " + JSON.stringify(number));
    return "";
  }
  
  try {
    var clean = String(number).replace(/\D/g, '');
    if (!clean) return "";
    
    if (clean.startsWith("0")) return "62" + clean.substring(1);
    if (clean.startsWith("62")) return clean;
    if (clean.startsWith("8")) return "62" + clean;
    return clean;
  } catch (e) {
    Logger.log("Gagal memformat nomor: " + e.message);
    return "";
  }
}

/**
 * Validasi nomor WhatsApp dengan endpoint_check_number
 */
function isWhatsappNumberValid(number, config) {
  // Jika tidak ada endpoint untuk cek nomor, anggap nomor valid
  if (!config.endpoint_check_number) {
    return true;
  }

  var payload = {
    api_key: config.api_key,
    sender: config.sender,
    number: number
  };
  
  var options = {
    method: "post",
    contentType: "application/json",
    payload: JSON.stringify(payload),
    muteHttpExceptions: true
  };
  
  try {
    var response = UrlFetchApp.fetch(config.endpoint_check_number, options);
    var result = JSON.parse(response.getContentText());
    return result.status === true;
  } catch (err) {
    Logger.log("Error checking WhatsApp number: " + err.message);
    return false;
  }
}

/**
 * Parsing data pendaftaran dari pesan
 */
function parseRegistrationData(message) {
  // Coba format standar terlebih dahulu
  var data = {
    name: extractData(message, /nama:\s*(.*?)(?=jenis kelamin:|$)/i, true) || "",
    jenisKelamin: extractData(message, /jenis kelamin:\s*(.*?)(?=email:|$)/i, true) || "",
    email: extractData(message, /email:\s*(.*?)(?=nomor wa:|$)/i, false) || "",
    number: extractData(message, /nomor wa:\s*(.*?)(?=nik:|$)/i, false) || "",
    nik: extractData(message, /nik:\s*(.*?)(?=nisn:|$)/i, false) || "",
    nisn: extractData(message, /nisn:\s*(.*?)(?=asal sekolah:|$)/i, false) || "",
    asalSekolah: extractData(message, /asal sekolah:\s*(.*?)$/i, true) || ""
  };
  
  // Jika format standar tidak ditemukan, coba format alternatif (satu baris)
  if (!data.name && !data.number) {
    var lines = message.split('\n');
    if (lines.length >= 7) {
      data = {
        name: lines[0].replace(/^nama:\s*/i, '').trim(),
        jenisKelamin: lines[1].replace(/^jenis kelamin:\s*/i, '').trim(),
        email: lines[2].replace(/^email:\s*/i, '').trim(),
        number: lines[3].replace(/^nomor wa:\s*/i, '').trim(),
        nik: lines[4].replace(/^nik:\s*/i, '').trim(),
        nisn: lines[5].replace(/^nisn:\s*/i, '').trim(),
        asalSekolah: lines[6].replace(/^asal sekolah:\s*/i, '').trim()
      };
    }
  }
  
  return data;
}

/**
 * Ekstrak data dari teks menggunakan regex
 */
function extractData(text, regex, toUpper) {
  var match = text.match(regex);
  if (!match) return "";
  var value = match[1].trim();
  return toUpper ? value.toUpperCase() : value.toLowerCase();
}

/**
 * Cari data duplikat berdasarkan nomor WA atau NIK
 */
function findDuplicateData(sheet, number, nik) {
  if ((!number && !nik) || sheet.getLastRow() < 2) return null;
  
  var data = sheet.getRange(2, 1, sheet.getLastRow()-1, 10).getValues();
  for (var i = 0; i < data.length; i++) {
    var existingNumber = formatNomorIndonesia(data[i][4]); // Kolom NOMOR WHATSAPP sekarang di index 4
    var existingNik = data[i][7] ? data[i][7].toString().replace(/^'/, '') : ''; // Kolom NIK sekarang di index 7
    
    // Cek duplikasi nomor WA
    if (number && existingNumber === number) {
      return {
        type: 'number',
        row: i + 2,
        name: data[i][3], // Kolom NAMA LENGKAP sekarang di index 3
        jenisKelamin: data[i][5], // Kolom JENIS KELAMIN sekarang di index 5
        email: data[i][6], // Kolom EMAIL sekarang di index 6
        number: existingNumber,
        nik: existingNik,
        nisn: data[i][8], // Kolom NISN sekarang di index 8
        asalSekolah: data[i][9], // Kolom ASAL SEKOLAH sekarang di index 9
        jenisAfirmasi: data[i][2] // Kolom JENIS AFIRMASI sekarang di index 2
      };
    }
    
    // Cek duplikasi NIK (jika NIK diinput)
    if (nik && existingNik === nik.replace(/^'/, '')) {
      return {
        type: 'nik',
        row: i + 2,
        name: data[i][3],
        jenisKelamin: data[i][5],
        email: data[i][6],
        number: existingNumber,
        nik: existingNik,
        nisn: data[i][8],
        asalSekolah: data[i][9],
        jenisAfirmasi: data[i][2]
      };
    }
  }
  return null;
}

/**
 * Simpan data ke sheet Pendaftaran dengan struktur baru
 */
function saveToSheet(sheet, name, email, number, jenisKelamin, nik, nisn, asalSekolah, jenisAfirmasi, config) {
  // Buat header jika sheet kosong
  if (sheet.getLastRow() === 0) {
    sheet.appendRow([
      "TIMESTAMP", 
      "NOMOR PENDAFTARAN", 
      "JENIS AFIRMASI",
      "NAMA LENGKAP", 
      "NOMOR WHATSAPP", 
      "JENIS KELAMIN",
      "EMAIL", 
      "NIK", 
      "NISN", 
      "ASAL SEKOLAH"
    ]);
    
    // Set format kolom NIK dan NISN sebagai teks
    sheet.getRange("H2:H").setNumberFormat("@");
    sheet.getRange("I2:I").setNumberFormat("@");
  }
  
  // Hitung nomor urut (baris saat ini dikurangi 1 karena header)
  var rowNumber = sheet.getLastRow() - 1;
  
  // Generate nomor peserta jika format tersedia
  var generatedParticipant = "";
  if (config && config.format_pendaftaran) {
    generatedParticipant = config.format_pendaftaran.replace("${nomor}", rowNumber);
  }
  
  // Simpan data dengan struktur baru
  sheet.appendRow([
    new Date(), 
    generatedParticipant, 
    jenisAfirmasi || "",
    name, 
    number, 
    jenisKelamin,
    email, 
    nik, 
    nisn, 
    asalSekolah
  ]);
  
  return generatedParticipant;
}
